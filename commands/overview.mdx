## Moderation

<CardGroup cols="2">
  <Card title="Timeout" icon="clock" iconType="solid" color="#396cc9" href="https://docs.prismboat.com/commands/code">
    \!timeout (user) (duration) \<reason\>

    \!timeout @mxybecookie 7d spamming
  </Card>
  <Card title="Remove Timeout" icon="clock" iconType="solid" color="#396cc9">
    \!removetimeout (user) \<reason\>

    \!removetimeout @mxybecookie idk
  </Card>
  <Card title="Purge" icon="trash" iconType="solid" color="#396cc9">
    \!purge (amount)

    \!purge 99
  </Card>
  <Card title="Setnick" icon="paintbrush" iconType="solid" color="#396cc9">
    \!setnick (user) (nickname)

    \!setnick @mxybecookie cookie
  </Card>
  <Card title="Role" icon="User" iconType="solid" color="#396cc9">
    \!role (user) (role)\
    \!role @mxybecookie founder
  </Card>
  <Card title="Kick" icon="minus" iconType="solid" color="#396cc9">
    \!kick (user) \<reason\>

    \!kick @mxybecookie being bad
  </Card>
  <Card title="Ban" icon="hammer" iconType="solid" color="#396cc9">
    \!ban (user) \<reason\>

    \!ban @mxybecookie breaking server rules
  </Card>
</CardGroup>
